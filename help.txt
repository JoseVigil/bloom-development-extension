│                                                                                                  │
│  COMPRESS - Compress files into optimized JSON for AI consumption                                │
│                                                                                                  │
│    python -m brain filesystem compress <PATHS> [OPTIONS]                                         │
│                                                                                                  │
│    Argumentos:                                                                                   │
│      <PATHS>              Paths to compress (files or directories)                               │
│                                                                                                  │
│    Opciones:                                                                                     │
│      -m                   Compression mode: 'codebase' or 'docbase'                              │
│      -o                   Output directory (default: current directory)                          │
│      -e                   Patterns to exclude (can be used multiple times)                       │
│      --no-comments        Remove comments during compression                                     │
│                                                                                                  │
│  EXTRACT - Extract files from compressed JSON format                                             │
│                                                                                                  │
│    python -m brain filesystem extract <JSON_FILE> [OPTIONS]                                      │
│                                                                                                  │
│    Argumentos:                                                                                   │
│      <JSON_FILE>          Compressed JSON file to extract                                        │
│                                                                                                  │
│    Opciones:                                                                                     │
│      -o                   Output directory (default: extracted_<mode>)                           │
│      --verify-hashes/--no-verify-hashes Verify MD5 hashes during extraction                      │
│                                                                                                  │
│  TREE - Generate visual tree structure with optional MD5 hashing and JSON export                 │
│                                                                                                  │
│    python -m brain filesystem tree <TARGETS> [OPTIONS]                                           │
│                                                                                                  │
│    Argumentos:                                                                                   │
│      <TARGETS>            Specific directories or files to include.                              │
│                                                                                                  │
│    Opciones:                                                                                     │
│      -o                   Output file path                                                       │
│      --hash               Include MD5 hashes                                                     │
│      --export-json        Export JSON to disk                                                    │
│                                                                                                  │
